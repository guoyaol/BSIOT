全部是post请求

http://localhost:8080/register?username=abc2&password=ab2c&email=abc2		//创建用户
http://localhost:8080/login json={username:test,password:test}				//登录，除了注册其他接口都要求先登录
http://localhost:8080/checklogin						//检查是否登录

http://localhost:8080/createdevice?clientid=test&name=test&description=test		//添加设备
http://localhost:8080/modifydevice?clientid=test&name=test2				//修改设备名称
http://localhost:8080/deletedevice?clientid=test					//删除设备

http://localhost:8080/gettotalamount						//总行数
http://localhost:8080/getdeviceamount						//设备数量
http://localhost:8080/gethistory?clientid=device0001				//获得设备历史信息（lan、lng）
http://localhost:8080/getallinfo?clientid=device0001					//获得设备的所有信息
http://localhost:8080/getlatestalert						//获得最新警报
